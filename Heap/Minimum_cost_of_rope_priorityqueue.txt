long long minCost(long long arr[], long long n) {
        priority_queue <long long, vector<long long>, greater<long long> > pq;
        long long cost = 0;
        for(long long i = 0; i < n; i++)
        {
            pq.push(arr[i]);
        }
        while(pq.size() != 1)
        {
            long long n1 = pq.top();
            pq.pop();
            long long n2 = pq.top();
            pq.pop();
            cost += n1+n2;
            pq.push(n1+n2);
        }
        return cost;
    }
